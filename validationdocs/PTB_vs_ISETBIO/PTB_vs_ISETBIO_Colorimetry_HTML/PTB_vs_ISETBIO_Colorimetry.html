function [validationReport, validationFailedFlag, validationDataToSave] = validationScript(runParams)

Initialize return params

    validationReport = '';
    validationFailedFlag = false;
    validationDataToSave = struct();

SETUP

    isetbioPath = fileparts(which('colorTransformMatrix'));
    curDir = pwd;

XYZ-related colorimetry

xy

    message = sprintf('\n***** Basic XYZ *****\n');
    validationReport = sprintf('%s %s', validationReport, message);

    testXYZs = [[1 2 1]' [2 1 0.5]' [1 1 1]' [0.6 2.3 4]'];
    ptbxyYs = XYZToxyY(testXYZs);
    ptbxys  = ptbxyYs(1:2,:);
    isetxys = chromaticity(testXYZs')';

    tolerance = 1e-10;
    if (any(abs(ptbxys-isetxys) > 1e-10))
        message = sprintf('PTB-ISET DIFFERENCE for XYZ to xy (tolerance: %g) \n', tolerance);
    else
        message = sprintf('PTB-ISET AGREE for XYZ to xy (tolerance: %g)\n', tolerance);
    end

    % update validation report
    validationReport = sprintf('%s %s', validationReport, message);

    % update validationDataToSave struct
    validationDataToSave.testXYZs = testXYZs;
    validationDataToSave.ptbxys = ptbxys;
    validationDataToSave.isetxys = isetxys;

xyY

    ptbXYZs = xyYToXYZ(ptbxyYs);
    tolerance = 1e-10;
    if (any(abs(testXYZs-ptbXYZs) > tolerance))
        message = sprintf('PTB FAILS XYZ to xyY to XYZ (tolerance: %g)\n', tolerance);
    else
        message = sprintf('PTB PASSES XYZ to xyY to XYZ (tolerance: %g)\n', tolerance);
    end

    % update validation report and validationDataToSave struct
    validationReport = sprintf('%s %s', validationReport, message);
    validationDataToSave.ptbXYZs = ptbXYZs;

    isetXYZs = xyy2xyz(ptbxyYs')';
    if (any(abs(testXYZs-isetXYZs) > tolerance))
        message = sprintf('PTB-ISET DIFFERENCE for xyY to XYZ (tolerance: %g)\n', tolerance);
    else
        message = sprintf('PTB-ISET AGREE for xyY to XYZ (tolerance: %g)\n', tolerance);
    end

    % update validation report and validationDataToSave struct
    validationReport = sprintf('%s %s', validationReport, message);
    validationDataToSave.isetXYZs = isetXYZs;

CIE uv chromaticity

    ptbuvYs = XYZTouvY(testXYZs);
    ptbuvs = ptbuvYs(1:2,:);
    [isetus,isetvs] = xyz2uv(testXYZs');
    isetuvs = [isetus' ; isetvs'];
    if (any(abs(ptbuvs-isetuvs) > 1e-10))
        message = sprintf('PTB-ISET DIFFERENCE for XYZ to uv\n');
        message = printf('%s\tI think this is because ISET implements an obsolete version of the standard\n', message);
    else
        message = sprintf('PTB-ISET AGREE for XYZ to uv\n');
    end
    % update validation report and validationDataToSave struct
    validationReport = sprintf('%s %s', validationReport, message);
    validationDataToSave.ptbuvs = ptbuvs;
    validationDataToSave.isetuvs = isetuvs;
